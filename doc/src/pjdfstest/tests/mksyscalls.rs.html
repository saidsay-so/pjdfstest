19/// Assert that the created entry gets its permission bits from the mode
20/// provided to the function negated by the process's file creation mask
21/// (umask), and its file type is equal to the expected one.
22pub(super) fn assert_perms_from_mode_and_umask<F, T, C>(
23    ctx: &mut SerializedTestContext,
24    f: F,
25    f_type_check: C,
26) where
27    F: Fn(&Path, Mode) -> nix::Result<T>,
28    C: Fn(&FileType) -> bool,
29{
30    fn assert_perm<F, T, C>(
31        ctx: &SerializedTestContext,
32        mode: mode_t,
33        expected_mode: mode_t,
34        f: F,
35        f_type_check: C,
36    ) where
37        F: Fn(&Path, Mode) -> nix::Result<T>,
38        C: Fn(&FileType) -> bool,
39    {
40        let path = ctx.gen_path();
41        assert!(f(&path, Mode::from_bits_truncate(mode)).is_ok());
42        let meta = metadata(&path).unwrap();
43        assert!(f_type_check(&meta.file_type()));
44        assert_eq!(
45            meta.permissions().mode() as mode_t & ALLPERMS,
46            expected_mode
47        );
48    }
49
50    /// Assert that the created entry permission bits equal `mode AND (NOT umask)`.
51    fn assert_perm_umask<F, T, C>(
52        ctx: &SerializedTestContext,
53        mode: mode_t,
54        umask: mode_t,
55        f: F,
56        check: C,
57    ) where
58        F: Fn(&Path, Mode) -> nix::Result<T>,
59        C: Fn(&FileType) -> bool,
60    {
61        ctx.with_umask(umask, || {
62            assert_perm(ctx, mode, mode & (!umask), f, check);
63        })
64    }
65
66    /// Assert that the created entry permission bits equal mode.
67    fn assert_perm_mode<F, T, C>(ctx: &SerializedTestContext, mode: mode_t, f: F, check: C)
68    where
69        F: Fn(&Path, Mode) -> nix::Result<T>,
70        C: Fn(&FileType) -> bool,
71    {
72        assert_perm(ctx, mode, mode, f, check);
73    }
74
75    assert_perm_mode(ctx, 0o755, &f, &f_type_check);
76    assert_perm_mode(ctx, 0o151, &f, &f_type_check);
77    assert_perm_umask(ctx, 0o151, 0o77, &f, &f_type_check);
78    assert_perm_umask(ctx, 0o345, 0o70, &f, &f_type_check);
79    assert_perm_umask(ctx, 0o501, 0o345, f, f_type_check);
80}

82/// Assert that the entry's user ID is set to the process' effective user ID and
83/// the entry's group ID should be set to the group ID of the parent directory
84/// or the effective group ID of the process.
85pub(super) fn assert_uid_gid<F, T>(ctx: &mut SerializedTestContext, f: F)
86where
87    F: Fn(&Path, Mode) -> nix::Result<T>,
88{
89    fn doit<F, T>(ctx: &SerializedTestContext, user: &User, gid: Option<Gid>, f: F)
90    where
91        F: Fn(&Path, Mode) -> nix::Result<T>,
92    {
93        let path = ctx.gen_path();
94        ctx.as_user(user, gid.map(|g| vec![g]).as_deref(), || {
95            f(&path, Mode::from_bits_truncate(0o755)).unwrap();
96        });
97
98        let filestat = lstat(&path).unwrap();
99        assert_eq!(filestat.st_uid, user.uid.as_raw());
100
101        let egid = gid.unwrap_or(user.gid).as_raw();
102        let dirstat = lstat(ctx.base_path()).unwrap();
103        assert!(filestat.st_gid == egid || filestat.st_gid == dirstat.st_gid);
104    }
105
106    let user = User::from_uid(Uid::effective()).unwrap().unwrap();
107    doit(ctx, &user, None, &f);
108
109    let user = ctx.get_new_user();
110    // To check that the entry gid is either parent gid or egid
111    chown(ctx.base_path(), Some(user.uid), Some(user.gid)).unwrap();
112
113    let (other_user, group) = ctx.get_new_entry();
114    doit(ctx, user, Some(group.gid), &f);
115
116    chmod(ctx.base_path(), Mode::from_bits_truncate(ALLPERMS)).unwrap();
117
118    let group = ctx.get_new_group();
119    doit(ctx, other_user, Some(group.gid), f);
120}
