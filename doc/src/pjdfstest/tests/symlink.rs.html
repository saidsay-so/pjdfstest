21crate::test_case! {
22    /// symlink creates symbolic links
23    // symlink/00.t
24    create_symlink => [Regular, Dir, Block, Char, Fifo]
25}
26fn create_symlink(ctx: &mut TestContext, ft: FileType) {
27    let file = ctx.create(ft.clone()).unwrap();
28    let link = ctx.gen_path();
29    assert!(symlink(&file, &link).is_ok());
30
31    let link_stat = symlink_metadata(&link).unwrap();
32    let follow_link_stat = metadata(&link).unwrap();
33    let follow_link_type = follow_link_stat.file_type();
34    assert!(link_stat.is_symlink());
35    assert!(match ft {
36        FileType::Regular => follow_link_type.is_file(),
37        FileType::Dir => follow_link_type.is_dir(),
38        FileType::Block => follow_link_type.is_block_device(),
39        FileType::Char => follow_link_type.is_char_device(),
40        FileType::Fifo => follow_link_type.is_fifo(),
41        _ => unreachable!(),
42    });
43
44    match ft {
45        FileType::Dir => remove_dir(&file),
46        _ => remove_file(&file),
47    }
48    .unwrap();
49    assert!(!link.exists());
50}

52crate::test_case! {
53    /// symlink create a symbolic link to a symbolic link
54    // symlink/00.t
55    create_symlink_to_symlink
56}
57fn create_symlink_to_symlink(ctx: &mut TestContext) {
58    let target = ctx.create(FileType::Regular).unwrap();
59    let file = ctx.create(FileType::Symlink(Some(target))).unwrap();
60    let link = ctx.gen_path();
61    assert!(symlink(&file, &link).is_ok());
62
63    let link_stat = symlink_metadata(&link).unwrap();
64    let follow_link_stat = metadata(&link).unwrap();
65    let follow_link_type = follow_link_stat.file_type();
66    assert!(link_stat.is_symlink());
67    assert!(follow_link_type.is_file());
68
69    remove_file(&file).unwrap();
70    assert!(!link.exists());
71}
