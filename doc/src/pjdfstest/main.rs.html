1//! This is the main entry point for the test suite. It is responsible for parsing
2//! command line arguments, reading the configuration file, and running the tests.
3//! The test suite is composed of a set of test cases, each of which is a function
4//! that takes a [`TestContext`] as an argument. The [`TestContext`] provides access to
5//! the configuration, the dummy authentication entries, and a temporary directory
6//! for the test to use.
7//!
8//! The test suite is built using the `inventory` crate, which
9//! allows test cases to be registered at compile time. The test suite is run by
10//! iterating over the registered test cases and running each one in turn.
11//!
12//! The [`TestContext`] is created for each test case, and the test case function is called
13//! with the [`TestContext`] as an argument. The test case function can then use the
14//! [`TestContext`] to access the configuration, the dummy authentication entries, and
15//! the temporary directory. The test case function can perform whatever tests are
16//! necessary, and panic if the test fails. The test suite catches the panic, prints
17//! an error message, and continues running the remaining test cases. At the end of
18//! the test suite, the number of failed, skipped, and passed tests is printed.
19

88#[derive(Clone, Copy, Debug, Default)]
89struct OverallResult {
90    pass: usize,
91    fail: usize,
92    skip: usize,
93    expect_fail: usize,
94    unexpect_pass: usize,
95}
96
97impl OverallResult {
98    fn color(&self) -> Color {
99        if self.pass() {
100            Color::Green
101        } else {
102            Color::Red
103        }
104    }
105
106    fn pass(&self) -> bool {
107        self.fail + self.unexpect_pass == 0
108    }
109
110    fn total(&self) -> usize {
111        self.pass + self.fail + self.skip + self.expect_fail + self.unexpect_pass
112    }
113}

115fn main() -> anyhow::Result<()> {
116    let args = ArgOptions::parse();
117
118    if args.list_features {
119        for feature in FileSystemFeature::iter() {
120            println!("{feature}: {}", feature.get_documentation().unwrap());
121        }
122        return Ok(());
123    }
124
125    let mut config = args
126        .configuration_file
127        .as_ref()
128        .map(Config::load)
129        .unwrap_or_default();
130    if let Some(sfs) = args.secondary_fs {
131        config.features.secondary_fs = Some(sfs);
132    }
133
134    let path = args
135        .path
136        .ok_or_else(|| anyhow::anyhow!("cannot get current dir"))
137        .or_else(|_| current_dir())?;
138    let base_dir = tempdir_in(path)?;
139
140    set_hook(Box::new(|_| {
141        *BACKTRACE.lock().unwrap() = Some(Backtrace::capture());
142    }));
143
144    let test_cases = inventory::iter::<TestCase>;
145    let test_cases: Vec<_> = test_cases
146        .into_iter()
147        .filter(|case| {
148            args.test_patterns.is_empty()
149                || args.test_patterns.iter().any(|pat| {
150                    if args.exact {
151                        case.name == pat
152                    } else {
153                        case.name.contains(pat)
154                    }
155                })
156        })
157        .map(|tc: &TestCase| TestCase {
158            // Ideally trim_start_matches could be done in test_case!, but only
159            // const functions are allowed there.
160            name: tc.name.trim_start_matches("pjdfstest::tests::"),
161            description: tc.description,
162            require_root: tc.require_root,
163            fun: tc.fun,
164            required_features: tc.required_features,
165            guards: tc.guards,
166        })
167        .collect();
168
169    umask(Mode::empty());
170
171    let overall_result = run_test_cases(&test_cases, args.verbose, &config, base_dir)?;
172
173    println!(
174        "\n{}: {} {}, {} {}, {} {}, {} {}, {} total",
175        "Summary".color(overall_result.color()).bold(),
176        (overall_result.fail + overall_result.unexpect_pass),
177        "failed".red().bold(),
178        overall_result.skip,
179        "skipped".yellow().bold(),
180        overall_result.pass,
181        "passed".green().bold(),
182        overall_result.expect_fail,
183        "expected failures".bright_green().bold(),
184        overall_result.total()
185    );
186
187    if overall_result.pass() {
188        Ok(())
189    } else {
190        Err(anyhow::anyhow!("Some tests have failed"))
191    }
192}

194/// Run provided test cases and filter according to features and flags availability.
195//TODO: Refactor this function
196fn run_test_cases(
197    test_cases: &[TestCase],
198    verbose: bool,
199    config: &Config,
200    base_dir: TempDir,
201) -> Result<OverallResult, anyhow::Error> {
202    let mut failed_tests_count: usize = 0;
203    let mut succeeded_tests_count: usize = 0;
204    let mut skipped_tests_count: usize = 0;
205    let mut unexpected_success_count: usize = 0;
206    let mut expected_fail_count: usize = 0;
207
208    let is_root = Uid::current().is_root();
209
210    let enabled_features: HashSet<_> = config.features.fs_features.keys().collect();
211
212    let entries = &config.dummy_auth.entries;
213
214    for test_case in test_cases {
215        //TODO: There's probably a better way to do this...
216        let mut should_skip = test_case.require_root && !is_root;
217        let mut skip_reasons = Vec::<String>::new();
218
219        let expect_fail = config.settings.expected_failures.contains(test_case.name);
220        if should_skip {
221            skip_reasons.push(String::from("requires root privileges"));
222        }
223
224        let features: HashSet<_> = test_case.required_features.iter().collect();
225        let missing_features: Vec<_> = features.difference(&enabled_features).collect();
226        if !missing_features.is_empty() {
227            should_skip = true;
228
229            let features = &missing_features
230                .iter()
231                .map(|feature| format!("{}", feature))
232                .collect::<Vec<_>>()
233                .join(", ");
234
235            skip_reasons.push(format!("requires features: {}", features));
236        }
237
238        let temp_dir = tempdir_in(base_dir.path()).unwrap();
239        // FIX: some tests need a 0o755 base dir
240        chmod(temp_dir.path(), Mode::from_bits_truncate(0o755)).unwrap();
241
242        if test_case
243            .guards
244            .iter()
245            .any(|guard| guard(config, temp_dir.path()).is_err())
246        {
247            should_skip = true;
248            skip_reasons.extend(
249                test_case
250                    .guards
251                    .iter()
252                    .filter_map(|guard| guard(config, base_dir.path()).err())
253                    .map(|err| err.to_string()),
254            );
255        }
256
257        stdout().lock().flush()?;
258
259        if should_skip {
260            println!("{:72} {}", test_case.name.blue(), "skipped".yellow());
261            if verbose && !test_case.description.is_empty() {
262                println!("\t{}", test_case.description);
263            }
264            for reason in &skip_reasons {
265                println!("\t{}", reason.yellow());
266            }
267            skipped_tests_count += 1;
268            continue;
269        }
270
271        let result = catch_unwind(|| match test_case.fun {
272            TestFn::NonSerialized(fun) => {
273                let mut context = TestContext::new(config, entries, temp_dir.path());
274
275                (fun)(&mut context)
276            }
277            TestFn::Serialized(fun) => {
278                let mut context = SerializedTestContext::new(config, entries, temp_dir.path());
279
280                (fun)(&mut context)
281            }
282        });
283
284        let error_info = match result {
285            Ok(_) if !expect_fail => {
286                println!("{:77} {}", test_case.name.blue(), "ok".green());
287                succeeded_tests_count += 1;
288                None
289            }
290            Ok(_) => {
291                println!(
292                    "{:60} {}",
293                    test_case.name.blue(),
294                    "PASSED UNEXPECTEDLY".red()
295                );
296                unexpected_success_count += 1;
297                None
298            }
299            Err(_) if expect_fail => {
300                println!(
301                    "{:61} {}",
302                    test_case.name.blue(),
303                    "failed as expected".green()
304                );
305                expected_fail_count += 1;
306                None
307            }
308            Err(e) => {
309                let backtrace = BACKTRACE
310                    .lock()
311                    .unwrap()
312                    .take()
313                    .filter(|bt| bt.status() == BacktraceStatus::Captured);
314                let panic_information = match e.downcast::<String>() {
315                    Ok(v) => *v,
316                    Err(e) => match e.downcast::<&str>() {
317                        Ok(v) => v.to_string(),
318                        _ => "Unknown Source of Error".to_owned(),
319                    },
320                };
321                println!("{:73} {}", test_case.name.blue(), "FAILED".red());
322                failed_tests_count += 1;
323                Some((panic_information, backtrace))
324            }
325        };
326        if verbose && !test_case.description.is_empty() {
327            println!("\t{}", test_case.description);
328        }
329        if let Some((panic_information, backtrace)) = error_info {
330            println!("\t{}", panic_information);
331            if let Some(backtrace) = backtrace {
332                println!("Backtrace:\n{}", backtrace);
333            }
334        }
335
336        if verbose && !test_case.description.is_empty() {
337            println!("\t{}", test_case.description);
338        }
339    }
340
341    Ok(OverallResult {
342        fail: failed_tests_count,
343        skip: skipped_tests_count,
344        pass: succeeded_tests_count,
345        unexpect_pass: unexpected_success_count,
346        expect_fail: expected_fail_count,
347    })
348}
