1//! This is the main entry point for the test suite. It is responsible for parsing
2//! command line arguments, reading the configuration file, and running the tests.
3//! The test suite is composed of a set of test cases, each of which is a function
4//! that takes a [`TestContext`] as an argument. The [`TestContext`] provides access to
5//! the configuration, the dummy authentication entries, and a temporary directory
6//! for the test to use.
7//!
8//! The test suite is built using the `inventory` crate, which
9//! allows test cases to be registered at compile time. The test suite is run by
10//! iterating over the registered test cases and running each one in turn.
11//!
12//! The [`TestContext`] is created for each test case, and the test case function is called
13//! with the [`TestContext`] as an argument. The test case function can then use the
14//! [`TestContext`] to access the configuration, the dummy authentication entries, and
15//! the temporary directory. The test case function can perform whatever tests are
16//! necessary, and panic if the test fails. The test suite catches the panic, prints
17//! an error message, and continues running the remaining test cases. At the end of
18//! the test suite, the number of failed, skipped, and passed tests is printed.
19

62#[derive(Debug, Options)]
63struct ArgOptions {

65    help: bool,

70    #[options(help = "List opt-in features")]
71    list_features: bool,

73    #[options(help = "Match names exactly")]
74    exact: bool,
75
76    #[options(help = "Verbose mode")]
77    verbose: bool,
78
79    #[options(help = "Path where the test suite will be executed")]
80    path: Option<PathBuf>,

83    test_patterns: Vec<String>,

86    secondary_fs: Option<PathBuf>,

89#[derive(Clone, Copy, Debug, Default)]
90struct OverallResult {
91    pass: usize,
92    fail: usize,
93    skip: usize,
94    expect_fail: usize,
95    unexpect_pass: usize,
96}
97
98impl OverallResult {
99    fn pass(&self) -> bool {
100        self.fail + self.unexpect_pass == 0
101    }
102
103    fn total(&self) -> usize {
104        self.pass + self.fail + self.skip + self.expect_fail + self.unexpect_pass
105    }
106}

108fn main() -> anyhow::Result<()> {
109    let args = ArgOptions::parse_args_default_or_exit();
110
111    if args.list_features {
112        for feature in FileSystemFeature::iter() {
113            println!("{feature}: {}", feature.get_documentation().unwrap());
114        }
115        return Ok(());
116    }
117
118    let config: Config = {
119        let mut figment = Figment::from(Serialized::defaults(Config::default()));
120        if let Some(path) = args.configuration_file.as_deref() {
121            figment = figment.merge(Toml::file(path))
122        }
123
124        let mut config: Config = figment.extract()?;
125        config.features.secondary_fs = args.secondary_fs;
126        config
127    };
128
129    let path = args
130        .path
131        .ok_or_else(|| anyhow::anyhow!("cannot get current dir"))
132        .or_else(|_| current_dir())?;
133    let base_dir = tempdir_in(path)?;
134
135    set_hook(Box::new(|_| {
136        *BACKTRACE.lock().unwrap() = Some(Backtrace::capture());
137    }));
138
139    let test_cases = inventory::iter::<TestCase>;
140    let test_cases: Vec<_> = test_cases
141        .into_iter()
142        .filter(|case| {
143            args.test_patterns.is_empty()
144                || args.test_patterns.iter().any(|pat| {
145                    if args.exact {
146                        case.name == pat
147                    } else {
148                        case.name.contains(pat)
149                    }
150                })
151        })
152        .map(|tc: &TestCase| TestCase {
153            // Ideally trim_start_matches could be done in test_case!, but only
154            // const functions are allowed there.
155            name: tc.name.trim_start_matches("pjdfstest::tests::"),
156            description: tc.description,
157            require_root: tc.require_root,
158            fun: tc.fun,
159            required_features: tc.required_features,
160            guards: tc.guards,
161        })
162        .collect();
163
164    umask(Mode::empty());
165
166    let overall_result = run_test_cases(&test_cases, args.verbose, &config, base_dir)?;
167
168    println!(
169        "\nTests: {} failed, {} skipped, {} passed, {} expected failures {} total",
170        overall_result.fail + overall_result.unexpect_pass,
171        overall_result.skip,
172        overall_result.pass,
173        overall_result.expect_fail,
174        overall_result.total()
175    );
176
177    if overall_result.pass() {
178        Ok(())
179    } else {
180        Err(anyhow::anyhow!("Some tests have failed"))
181    }
182}

184/// Run provided test cases and filter according to features and flags availability.
185//TODO: Refactor this function
186fn run_test_cases(
187    test_cases: &[TestCase],
188    verbose: bool,
189    config: &Config,
190    base_dir: TempDir,
191) -> Result<OverallResult, anyhow::Error> {
192    let mut failed_tests_count: usize = 0;
193    let mut succeeded_tests_count: usize = 0;
194    let mut skipped_tests_count: usize = 0;
195    let mut unexpected_success_count: usize = 0;
196    let mut expected_fail_count: usize = 0;
197
198    let is_root = Uid::current().is_root();
199
200    let enabled_features: HashSet<_> = config.features.fs_features.keys().collect();
201
202    let entries = &config.dummy_auth.entries;
203
204    for test_case in test_cases {
205        //TODO: There's probably a better way to do this...
206        let mut should_skip = test_case.require_root && !is_root;
207        let mut skip_reasons = Vec::<String>::new();
208
209        let expect_fail = config.settings.expected_failures.contains(test_case.name);
210        if should_skip {
211            skip_reasons.push(String::from("requires root privileges"));
212        }
213
214        let features: HashSet<_> = test_case.required_features.iter().collect();
215        let missing_features: Vec<_> = features.difference(&enabled_features).collect();
216        if !missing_features.is_empty() {
217            should_skip = true;
218
219            let features = &missing_features
220                .iter()
221                .map(|feature| format!("{}", feature))
222                .collect::<Vec<_>>()
223                .join(", ");
224
225            skip_reasons.push(format!("requires features: {}", features));
226        }
227
228        let temp_dir = tempdir_in(base_dir.path()).unwrap();
229        // FIX: some tests need a 0o755 base dir
230        chmod(temp_dir.path(), Mode::from_bits_truncate(0o755)).unwrap();
231
232        if test_case
233            .guards
234            .iter()
235            .any(|guard| guard(config, temp_dir.path()).is_err())
236        {
237            should_skip = true;
238            skip_reasons.extend(
239                test_case
240                    .guards
241                    .iter()
242                    .filter_map(|guard| guard(config, base_dir.path()).err())
243                    .map(|err| err.to_string()),
244            );
245        }
246
247        // TODO: ;decide what to do about verbose
248        if verbose && !test_case.description.is_empty() {
249            print!("\n\t{}\t\t", test_case.description);
250        }
251
252        stdout().lock().flush()?;
253
254        if should_skip {
255            println!("{:72} skipped", test_case.name);
256            for reason in &skip_reasons {
257                println!("\t{}", reason);
258            }
259            skipped_tests_count += 1;
260            continue;
261        }
262
263        let result = catch_unwind(|| match test_case.fun {
264            TestFn::NonSerialized(fun) => {
265                let mut context = TestContext::new(config, entries, temp_dir.path());
266
267                (fun)(&mut context)
268            }
269            TestFn::Serialized(fun) => {
270                let mut context = SerializedTestContext::new(config, entries, temp_dir.path());
271
272                (fun)(&mut context)
273            }
274        });
275
276        match result {
277            Ok(_) if !expect_fail => {
278                println!("{:77} ok", test_case.name);
279                succeeded_tests_count += 1;
280            }
281            Ok(_) => {
282                println!("{:60} PASSED UNEXPECTEDLY", test_case.name);
283                unexpected_success_count += 1;
284            }
285            Err(_) if expect_fail => {
286                println!("{:61} failed as expected", test_case.name);
287                expected_fail_count += 1;
288            }
289            Err(e) => {
290                let backtrace = BACKTRACE
291                    .lock()
292                    .unwrap()
293                    .take()
294                    .filter(|bt| bt.status() == BacktraceStatus::Captured);
295                let panic_information = match e.downcast::<String>() {
296                    Ok(v) => *v,
297                    Err(e) => match e.downcast::<&str>() {
298                        Ok(v) => v.to_string(),
299                        _ => "Unknown Source of Error".to_owned(),
300                    },
301                };
302                println!("{:73} FAILED\n\t{}", test_case.name, panic_information);
303                if let Some(backtrace) = backtrace {
304                    println!("Backtrace:\n{}", backtrace);
305                }
306                failed_tests_count += 1;
307            }
308        }
309    }
310
311    Ok(OverallResult {
312        fail: failed_tests_count,
313        skip: skipped_tests_count,
314        pass: succeeded_tests_count,
315        unexpect_pass: unexpected_success_count,
316        expect_fail: expected_fail_count,
317    })
318}
